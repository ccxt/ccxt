class MyBinance(ccxt.binance):
    def parse_ohlcv(self, ohlcv, market=None):
        #

        return [
            self.safe_integer(ohlcv, 6),

            self.safe_number(ohlcv, 2),
            self.safe_number(ohlcv, 3),
            self.safe_number(ohlcv, 4),
            self.safe_number(ohlcv, 5),
        ]


exchange = MyBinance()
